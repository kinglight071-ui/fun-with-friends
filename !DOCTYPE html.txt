function genCode(){
  const chars = 'ABCDEFGHJKMNPQRSTUVWXYZ23456789';
  let out = '';
  for(let i=0;i<5;i++) out += chars[Math.floor(Math.random()*chars.length)];
  return out;
}

function linkForCode(code){
  return window.location.href.split('#')[0] + '#quiz=' + code;

async function saveQuiz(code, quiz){
  try{
    await db.collection('quizzes').doc(code).set(quiz);
    return true;
  }catch(e){ console.error('saveQuiz failed', e); return false; }
}
async function loadQuiz(code){
  try{
    const doc = await db.collection('quizzes').doc(code).get();
    return doc.exists ? doc.data() : null;
  }catch(e){ console.error('loadQuiz failed', e); return null; }
}
async function loadResults(code){
  try{
    const doc = await db.collection('results').doc(code).get();
    return doc.exists ? (doc.data().list || []) : [];
  }catch(e){ console.error('loadResults failed', e); return []; }
}
async function saveResults(code, results){
  try{

function useBankQuestion(index){
  const item = QUESTION_BANK[index];
  const filled = { ...blankQ(), q: item.q, correct: item.options[0], d1: item.options[1], d2: item.options[2], d3: item.options[3] };
  const empty = state.questions.find(q => !q.q.trim());
  if(empty){ Object.assign(empty, filled); }
  else { state.questions.push(filled); }
  render();
}

function addRandomQuestions(n){
  const usedTexts = new Set(state.questions.map(q => q.q.trim()).filter(Boolean));
  const pool = QUESTION_BANK.filter(item => !usedTexts.has(item.q));
  const picks = shuffleArray(pool).slice(0, n);
  const filledPicks = picks.map(item => ({ ...blankQ(), q: item.q, correct: item.options[0], d1: item.options[1], d2: item.options[2], d3: item.options[3] }));
  // drop any completely empty question blocks before adding the new batch
  state.questions = state.questions.filter(q => q.q.trim() || q.correct.trim() || q.d1.trim() || q.d2.trim() || q.d3.trim());
  state.questions.push(...filledPicks);
  if(state.questions.length === 0){ state.questions.push(blankQ()); }
  render();
}

function updateQ(i, field, val){ state.questions[i][field] = val; }

function removeQ(i){ state.questions.splice(i,1); render(); }

async function submitQuiz(){
  if(!state.name.trim()){ state.error = 'Add your name so friends know whose quiz this is.'; render(); return; }
  for(const q of state.questions){
    if(!q.q.trim() || !q.correct.trim() || !q.d1.trim() || !q.d2.trim() || !q.d3.trim()){
      state.error = 'Fill in every question, the correct answer, and all three decoys.';
      render();
      return;
    }
  }
  const code = genCode();
  const quiz = {
    creatorName: state.name.trim(),
    questions: state.questions.map(q => ({
      q: q.q.trim(),
      correct: { text: q.correct.trim(), img: q.correctImg.trim() },
      decoys: [
        { text: q.d1.trim(), img: q.d1Img.trim() },
        { text: q.d2.trim(), img: q.d2Img.trim() },
        { text: q.d3.trim(), img: q.d3Img.trim() }
      ]
    })),
    createdAt: Date.now()
  };
  const ok = await saveQuiz(code, quiz);
  if(!ok){ state.error = "Couldn't save the quiz — try again."; render(); return; }
  await saveResults(code, []);
  state = { screen:'created', code, quiz };
  render();
}

function copyLink(){

  navigator.clipboard.writeText(link).catch(()=>{});
  const btn = event.target;
  const old = btn.textContent;
  btn.textContent = 'Copied!';
  setTimeout(()=>{ btn.textContent = old; }, 1400);
}

async function loadQuizForTaking(codeOverride){
  const code = (codeOverride || state.code).trim().toUpperCase();
  if(!code){ state.error='Enter a code first.'; render(); return; }
  app.innerHTML = `<p class="loading">Looking up that quiz...</p>`;
  const quiz = await loadQuiz(code);
  if(!quiz){ state = { screen:'enterCode', code, error:"Couldn't find a quiz with that code. Double-check the link with your friend." }; render(); return; }
  const orderedQuestions = shuffleArray(quiz.questions);
  state = { screen:'takeQuiz', code, quiz, orderedQuestions, current:0, answers:[], takerName:'', pickedIdx:null, options: shuffleOptions(orderedQuestions[0]) };
  render();
}

function shuffleOptions(question){
  const opts = [question.correct, ...question.decoys].map((a,i)=>({text:a.text, img:a.img, isCorrect:i===0}));
  for(let i=opts.length-1;i>0;i--){
    const j = Math.floor(Math.random()*(i+1));
    [opts[i],opts[j]] = [opts[j],opts[i]];
  }
  return opts;
}

function shuffleArray(arr){
  const out = [...arr];
  for(let i=out.length-1;i>0;i--){
    const j = Math.floor(Math.random()*(i+1));
    [out[i],out[j]] = [out[j],out[i]];
  }
  return out;
}

function renderTakeQuiz(){
  const q = state.orderedQuestions[state.current];
  const total = state.orderedQuestions.length;

  if(state.current === 0 && !state.nameConfirmed){
    app.innerHTML = `
      <div class="eyebrow">${esc(state.quiz.creatorName)}'s quiz</div>
      <h1>How well do <em>you</em> know them?</h1>
      <div class="card">
        <label>Your name</label>
        <input type="text" placeholder="So they know who scored what" value="${esc(state.takerName)}" oninput="state.takerName=this.value">
        <p class="hint">Required — your name and score will be visible to everyone who takes this quiz.</p>
        ${state.nameError ? `<div class="error-msg">${esc(state.nameError)}</div>` : ''}
        <div class="btn-row">
          <button class="btn btn-primary" onclick="confirmName()">Begin</button>
        </div>
      </div>
    `;
    return;
  }

  app.innerHTML = `
    <div class="eyebrow">${esc(state.quiz.creatorName)}'s quiz</div>
    <div class="progress">Question ${state.current+1} of ${total}</div>
    <div class="card">
      <h1 style="font-size:1.5rem; margin-bottom:18px;">${esc(q.q)}</h1>
      <div id="optionsWrap">
        ${state.options.map((o,i)=>`
          <button class="answer-option" onclick="pickOption(${i})">
            ${o.img ? `<img src="${esc(o.img)}" onerror="this.style.display='none'">` : ''}
            <span>${esc(o.text)}</span>
          </button>
        `).join('')}
      </div>
    </div>
    <button class="small-link" onclick="viewLeaderboard('${state.code}')">See everyone's scores so far &rarr;</button>
  `;
}

function confirmName(){
  if(!state.takerName.trim()){ state.nameError = 'Enter your name to continue.'; render(); return; }
  state.nameError = '';
  state.nameConfirmed = true;
  render();
}

function pickOption(i){
  if(state.locked) return;
  state.locked = true;
  const chosen = state.options[i];
  state.answers.push({ question: state.orderedQuestions[state.current].q, chosenText: chosen.text, correct: chosen.isCorrect });

  const buttons = document.querySelectorAll('.answer-option');
  buttons.forEach((b, idx) => {
    b.style.pointerEvents = 'none';
    if(state.options[idx].isCorrect) b.classList.add('correct');
    else if(idx === i) b.classList.add('wrong');
  });

  setTimeout(()=>{
    state.locked = false;
    const next = state.current + 1;
    if(next < state.orderedQuestions.length){
      state.current = next;
      state.options = shuffleOptions(state.orderedQuestions[next]);
      render();
    } else {
      finishQuiz();
    }
  }, 900);
}

async function finishQuiz(){
  const score = state.answers.filter(a=>a.correct).length;
  const total = state.orderedQuestions.length;
  const results = await loadResults(state.code);
  const updated = [...results, { name: state.takerName.trim(), score, total, submittedAt: Date.now() }];
  await saveResults(state.code, updated);
  state = { screen:'results', code: state.code, quiz: state.quiz, results: updated, justSubmitted:true, myScore:score, myTotal:total, myName:state.takerName.trim(), answers: state.answers };
  render();
}

function renderResults(){
  const sorted = [...state.results].sort((a,b)=> (b.score/b.total) - (a.score/a.total));
  app.innerHTML = `
    <div class="eyebrow">${esc(state.quiz.creatorName)}'s quiz</div>
    <h1>${state.justSubmitted ? 'Nice try!' : 'Leaderboard'}</h1>
    ${state.justSubmitted ? `
      <div class="card score-hero">
        <div class="score-num">${state.myScore}/${state.myTotal}</div>
        <div class="score-label">${Math.round((state.myScore/state.myTotal)*100)}% — ${esc(state.myName)}'s score</div>
      </div>
    ` : ''}
    <div class="card">
      <label style="margin-top:0;">Who knows ${esc(state.quiz.creatorName)} best</label>
      ${sorted.length === 0 ? '<p class="hint">No one has taken this quiz yet.</p>' :
        sorted.map(r => `
          <div class="leader-row">
            <span class="leader-name">${esc(r.name)}</span>
            <span class="leader-score">${r.score}/${r.total} &middot; ${Math.round((r.score/r.total)*100)}%</span>
          </div>
        `).join('')
      }
    </div>
    <div class="btn-row">
      <button class="btn btn-secondary" onclick="goCreate()">Make your own quiz</button>
      <button class="btn btn-ghost" onclick="state={screen:'home'};render()">Back home</button>
    </div>
  `;
}
